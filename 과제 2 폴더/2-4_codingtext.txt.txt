#include <string>
#include <vector>

using namespace std;

int solution(int a, int b) {
    int answer = 0;
    
    int temp1 = stoi(to_string(a) + to_string(b));
    int temp2 = stoi(to_string(b) + to_string(a));

    if (temp1 > temp2)
        answer = temp1;
    else
        answer = temp2;
    
    return answer;
}
